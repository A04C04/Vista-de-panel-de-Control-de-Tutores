        // --- 2. LÓGICA DE NAVEGACIÓN Y VISTAS (SPA - Single Page Application) ---

        /**
         * Cambia la vista activa del dashboard.
         * @param {string} viewId - El ID de la sección a mostrar (ej: 'mis-reservas').
         * @param {string} title - El título a mostrar en el encabezado.
         */
        function navigateTo(viewId, title) {
            const views = document.querySelectorAll('.view-content');
            const navLinks = document.querySelectorAll('.nav-link');
            const viewTitle = document.getElementById('view-title');

            // Actualizar el título
            viewTitle.textContent = title;
            
            // Ocultar todas las vistas con animación
            views.forEach(view => {
                view.classList.add('view-hidden');
                view.classList.remove('block');
            });
            
            // Mostrar la vista seleccionada con animación
            setTimeout(() => {
                const activeView = document.getElementById(viewId);
                if (activeView) {
                    activeView.classList.remove('view-hidden');
                    activeView.classList.add('block'); // Asegura que se muestre
                    // Forzar el redibujado para que la transición funcione
                    activeView.offsetWidth; 
                    activeView.style.opacity = 1;
                    activeView.style.transform = 'translateY(0)';
                }
            }, 300); // 300ms de retraso para que la animación de ocultar termine

            // Actualizar el estado activo de los enlaces de navegación
            navLinks.forEach(link => {
                link.classList.remove('active');
                if (link.getAttribute('data-view') === viewId) {
                    link.classList.add('active');
                }
            });

             // Ocultar la barra lateral en móvil después de la navegación
            const sidebar = document.getElementById('sidebar');
            const overlay = document.getElementById('mobile-menu-overlay');
            if (sidebar.classList.contains('translate-x-0')) {
                 sidebar.classList.remove('translate-x-0');
                 sidebar.classList.add('-translate-x-full');
                 overlay.classList.add('hidden');
            }
        }

        // Manejar el cambio de hash en la URL
        function handleHashChange() {
            const hash = window.location.hash.substring(1); // Obtiene el hash sin el '#'
            let viewId = 'mis-reservas'; // Vista por defecto
            let title = 'Mis Reservas'; // Título por defecto

            if (hash) {
                const activeLink = document.querySelector(`.nav-link[href="#${hash}"]`);
                if (activeLink) {
                    viewId = activeLink.getAttribute('data-view');
                    title = activeLink.getAttribute('data-title');
                }
            }
            navigateTo(viewId, title);
        }

        // Escuchar el evento de cambio de hash
        window.addEventListener('hashchange', handleHashChange);

        // 3.2. Funcionalidad de Disponibilidad (Mi Disponibilidad)
        function populateTimeSelect(select, isEnabled) {
             // Lista de horas simplificada para el ejemplo
            const hours = [
                '07:00', '08:00', '09:00', '10:00', '11:00', '12:00',
                '13:00', '14:00', '15:00', '16:00', '17:00', '18:00',
                '19:00', '20:00', '21:00', '22:00'
            ];

            select.innerHTML = ''; // Limpiar opciones

            if (!isEnabled) {
                 select.innerHTML = `<option value="">-- No Disponible --</option>`;
                 return;
            }

            // Crear opciones de hora
            hours.forEach(hour => {
                const option = document.createElement('option');
                option.value = hour;
                option.textContent = hour;
                // Simulación: seleccionar una hora por defecto
                if (select.closest('tr').querySelector('input[type="checkbox"]').id === 'lunes-check' && select.parentElement.previousElementSibling.querySelector('select') === select && hour === '09:00') {
                    option.selected = true;
                } else if (select.closest('tr').querySelector('input[type="checkbox"]').id === 'lunes-check' && hour === '17:00') {
                     option.selected = true;
                }
                select.appendChild(option);
            });
        }

        document.addEventListener('DOMContentLoaded', () => {
             const videoUpload = document.getElementById('video-upload');
             const videoPreview = document.getElementById('video-preview');

             videoUpload.addEventListener('change', (event) => {
                 const file = event.target.files[0];
                 
                 if (!file) {
                     videoPreview.innerHTML = `<i data-lucide="upload" class="w-6 h-6 mx-auto mb-2"></i> Previsualización de video (ningún archivo seleccionado)`;
                     lucide.createIcons();
                     return;
                 }

                 if (file.type.startsWith('video/')) {
                     const fileURL = URL.createObjectURL(file);
                     
                     // Crear un elemento de video para la previsualización
                     const videoElement = document.createElement('video');
                     videoElement.src = fileURL;
                     videoElement.controls = true;
                     videoElement.className = 'w-full h-auto max-h-64 rounded-lg shadow-md mt-2';
                     
                     // Limpiar y añadir la previsualización
                     videoPreview.innerHTML = '';
                     videoPreview.classList.remove('border-dashed', 'text-gray-500');
                     videoPreview.classList.add('border-solid', 'p-0');
                     videoPreview.appendChild(videoElement);
                     
                 } else {
                     videoPreview.innerHTML = `<p class='text-center text-red-500'>⚠️ Archivo no válido. Por favor, sube un archivo de video.</p>`;
                     videoPreview.classList.add('border-dashed');
                 }
             });
         });

        document.addEventListener('DOMContentLoaded', () => {

            // 4.1. Lógica de los Checkboxes de Disponibilidad (Disponibilidad)
            document.querySelectorAll('.day-row input[type="checkbox"]').forEach(checkbox => {
                const row = checkbox.closest('.day-row');
                const selects = row.querySelectorAll('.time-select');
                const isInitialChecked = checkbox.checked;

                // Inicializar selectores al cargar
                selects.forEach(select => {
                    select.disabled = !isInitialChecked;
                    populateTimeSelect(select, isInitialChecked);
                });
                row.classList.toggle('bg-gray-100', !isInitialChecked);
                row.classList.toggle('bg-white', isInitialChecked);


                checkbox.addEventListener('change', function() {
                    const isChecked = this.checked;

                    selects.forEach(select => {
                        select.disabled = !isChecked;
                        // Rellena las opciones del select en función de si está disponible
                        populateTimeSelect(select, isChecked);
                    });

                    // Actualiza el estilo de la fila
                    row.classList.toggle('bg-gray-100', !isChecked);
                    row.classList.toggle('bg-white', isChecked);
                    row.classList.toggle('hover:bg-gray-200', !isChecked);
                    row.classList.toggle('hover:bg-gray-50', isChecked);
                });
            });

            // 4.2. Toggle del menú de perfil (Dropdown superior)
            const profileDropdown = document.getElementById('profile-dropdown');
            const btnToggleProfile = document.getElementById('btn-toggle-profile');

            btnToggleProfile.addEventListener('click', () => {
                const isHidden = profileDropdown.classList.contains('hidden');
                
                if (isHidden) {
                    profileDropdown.classList.remove('hidden');
                    // Muestra el dropdown con animación
                    setTimeout(() => {
                        profileDropdown.classList.remove('opacity-0', 'translate-y-2');
                    }, 10);
                } else {
                    // Oculta el dropdown con animación
                    profileDropdown.classList.add('opacity-0', 'translate-y-2');
                    setTimeout(() => {
                         profileDropdown.classList.add('hidden');
                    }, 200); // Coincide con la duración de la transición
                }
            });
            
            // 4.3. Toggle del menú lateral en móviles
            const sidebar = document.getElementById('sidebar');
            const btnToggleMobile = document.getElementById('btn-toggle-sidebar-mobile');
            const overlay = document.getElementById('mobile-menu-overlay');

            sidebar.classList.add('-translate-x-full'); // Estado inicial oculto en móvil

            btnToggleMobile.addEventListener('click', () => {
                sidebar.classList.toggle('-translate-x-full');
                sidebar.classList.toggle('translate-x-0');
                overlay.classList.toggle('hidden');
            });
            
            overlay.addEventListener('click', () => {
                sidebar.classList.add('-translate-x-full');
                sidebar.classList.remove('translate-x-0');
                overlay.classList.add('hidden');
            });

            document.getElementById('btn-desconectar-sidebar').addEventListener('click', Usuario.handleLogout);
            document.getElementById('btn-desconectar-dropdown').addEventListener('click', Usuario.handleLogout);

            handleHashChange();

            setTimeout(() => {
                if (typeof lucide !== 'undefined') {
                    lucide.createIcons();
                } else {
                    console.warn("Lucide Icons no se cargó correctamente.");
                }
            }, 100); 
        });
