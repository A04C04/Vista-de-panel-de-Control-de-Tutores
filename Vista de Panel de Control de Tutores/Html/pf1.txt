    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged, signOut } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-auth.js";
        import { getFirestore, doc, onSnapshot, collection, query, where, getDocs, updateDoc, setDoc, getDoc } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-firestore.js";
        import { setLogLevel } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-firestore.js";

        // Inicialización de variables globales
        let app, db, auth, userId = 'default-user';
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';
        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : null;
        const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : null;
        
        // Log de depuración para Firestore
        setLogLevel('debug');

        // Función de inicialización y autenticación
        async function initializeAndAuthenticate() {
            if (!firebaseConfig) {
                console.error("Firebase configuration is missing.");
                return;
            }
            try {
                app = initializeApp(firebaseConfig);
                db = getFirestore(app);
                auth = getAuth(app);
                
                if (initialAuthToken) {
                    await signInWithCustomToken(auth, initialAuthToken);
                } else {
                    await signInAnonymously(auth);
                }

                onAuthStateChanged(auth, (user) => {
                    if (user) {
                        userId = user.uid;
                        console.log("Usuario autenticado. UID:", userId);
                        document.getElementById('user-id-display').textContent = userId;
                        // Inicia la carga de datos y listeners una vez autenticado
                        loadInitialData(userId);
                    } else {
                        userId = 'anonymous';
                        console.log("Usuario anónimo o desautenticado.");
                        document.getElementById('user-id-display').textContent = 'Anónimo';
                        // Permite cargar datos públicos si los hubiera
                        loadInitialData(userId);
                    }
                });

            } catch (error) {
                console.error("Error en la inicialización o autenticación de Firebase:", error);
            }
        }

        // Simulación de módulo de usuario para desconexión
        const Usuario = {
            handleLogout: async () => {
                try {
                    await signOut(auth);
                    console.log("Sesión cerrada con éxito.");
                    // Opcional: Redirigir o actualizar la UI
                    // window.location.hash = '#login'; 
                } catch (error) {
                    console.error("Error al cerrar sesión:", error);
                }
            }
        };

        // Función para cargar datos iniciales (ejemplo)
        function loadInitialData(currentUserId) {
            // Ejemplo de listener para datos privados del tutor
            const tutorRef = doc(db, 'artifacts', appId, 'users', currentUserId, 'profile', 'data');
            onSnapshot(tutorRef, (docSnap) => {
                if (docSnap.exists()) {
                    const data = docSnap.data();
                    console.log("Datos del tutor:", data);
                    // Actualiza el nombre del tutor en la UI
                    document.getElementById('tutor-name-display').textContent = data.name || 'Tutor Desconocido';
                } else {
                    console.log("No hay datos de perfil para el tutor.");
                    // Crea un perfil por defecto si no existe
                    setDoc(tutorRef, { name: 'Tutor', bio: 'Bienvenido a tu panel.', lastLogin: new Date().toISOString() }, { merge: true })
                        .then(() => console.log("Perfil inicial creado."))
                        .catch(err => console.error("Error al crear perfil:", err));
                }
            }, (error) => {
                console.error("Error al escuchar cambios en el perfil:", error);
            });
            
            // Aquí puedes agregar listeners para otras colecciones (e.g., reservas, historial)
        }

        // Exportar funciones y variables necesarias para el scope global si es necesario
        window.db = db;
        window.auth = auth;
        window.Usuario = Usuario;
        window.initializeAndAuthenticate = initializeAndAuthenticate;
    </script>

        function renderView(viewName) {
            const view = views[viewName] || views['resumen'];
            document.getElementById('view-title').textContent = view.title;
            const container = document.getElementById('content-container');
            
            // Reemplaza el contenido, renderiza los íconos de Lucide
            container.innerHTML = view.html;

            // Actualiza los enlaces activos en el sidebar
            document.querySelectorAll('.sidebar-item').forEach(item => {
                item.classList.remove('active');
            });
            const activeLink = document.querySelector(`.sidebar-item[data-view="${viewName}"]`);
            if (activeLink) {
                activeLink.classList.add('active');
            }

            // Renderiza los nuevos íconos de Lucide en el contenido inyectado
            if (typeof lucide !== 'undefined') {
                lucide.createIcons();
            }

            // Cargar gráfico de ejemplo si estamos en la vista de resumen
            if (viewName === 'resumen') {
                createMonthlyChart();
            }
            
            // Sincroniza los nombres del tutor
            const tutorName = document.getElementById('tutor-name-display').textContent;
            document.getElementById('tutor-name-dropdown').textContent = tutorName;
            document.getElementById('tutor-name-summary').textContent = tutorName;
        }

        function handleHashChange() {
            const hash = window.location.hash.replace('#', '');
            renderView(hash);
        }

        // Lógica de Gráficos (Chart.js)
        function createMonthlyChart() {
            const ctx = document.getElementById('monthly-chart');
            if (ctx) {
                 // Destruir la instancia anterior si existe
                if (window.monthlyChartInstance) {
                    window.monthlyChartInstance.destroy();
                }

                window.monthlyChartInstance = new Chart(ctx, {
                    type: 'line',
                    data: {
                        labels: ['Ene', 'Feb', 'Mar', 'Abr', 'May', 'Jun', 'Jul'],
                        datasets: [{
                            label: 'Clases Impartidas',
                            data: [12, 19, 3, 5, 2, 3, 15],
                            borderColor: 'var(--secondary-color2)',
                            backgroundColor: 'rgba(33, 158, 188, 0.2)',
                            tension: 0.4,
                            fill: true,
                            borderWidth: 2,
                        }]
                    },
                    options: {
                        responsive: true,
                        maintainAspectRatio: false,
                        plugins: {
                            legend: {
                                display: false
                            }
                        },
                        scales: {
                            y: {
                                beginAtZero: true
                            }
                        }
                    }
                });
            }
        }

        function quickSearch() {
            const query = document.getElementById('search-input-modal').value.toLowerCase();
            const resultsContainer = document.getElementById('search-results');
            resultsContainer.innerHTML = '';

            if (query.length < 2) {
                resultsContainer.innerHTML = '<p class="text-gray-500 text-sm">Empieza a escribir para buscar secciones del panel...</p>';
                return;
            }

            const filteredResults = searchableViews.filter(view => 
                view.name.toLowerCase().includes(query) || view.keywords.includes(query)
            );

            if (filteredResults.length === 0) {
                resultsContainer.innerHTML = '<p class="text-gray-500 text-sm">No se encontraron resultados.</p>';
                return;
            }

            filteredResults.forEach(result => {
                const link = document.createElement('a');
                link.href = result.hash;
                link.textContent = result.name;
                link.className = 'block p-3 rounded-lg hover:bg-secondary-color/10 text-tertiary-color2 font-medium transition-colors';
                link.onclick = () => {
                    handleHashChange();
                    closeSearchModal();
                };
                resultsContainer.appendChild(link);
            });
        }

        document.addEventListener('DOMContentLoaded', () => {

            }

            // Toggle de la barra lateral
            const sidebar = document.getElementById('sidebar');
            const mainContent = document.getElementById('main-content');
            const toggleButton = document.getElementById('sidebar-toggle');
            const linkTexts = document.querySelectorAll('.link-text');
            const logoText = document.querySelector('.logo-text');

            toggleButton.addEventListener('click', () => {
                const isCollapsed = sidebar.classList.toggle('collapsed');
                mainContent.classList.toggle('collapsed', isCollapsed);
                toggleButton.querySelector('i').setAttribute('data-lucide', isCollapsed ? 'chevrons-right' : 'chevrons-left');
                logoText.classList.toggle('hidden', isCollapsed);

                linkTexts.forEach(span => {
                    span.classList.toggle('hidden', isCollapsed);
                });

                // Renderiza los íconos de Lucide nuevamente para actualizar el botón de toggle
                if (typeof lucide !== 'undefined') {
                    lucide.createIcons();
                }
            });

            // Toggle del menú de perfil
            const profileDropdown = document.getElementById('profile-dropdown');
            const btnToggleProfile = document.getElementById('btn-toggle-profile');

            btnToggleProfile.addEventListener('click', (e) => {
                e.stopPropagation(); // Evita que se cierre inmediatamente por el listener de document
                const isHidden = profileDropdown.classList.contains('hidden');
                
                if (isHidden) {
                    profileDropdown.classList.remove('hidden');
                    // Muestra el dropdown con animación
                    setTimeout(() => {
                        profileDropdown.classList.remove('opacity-0', 'translate-y-2');
                    }, 10);
                } else {
                    // Oculta el dropdown con animación
                    profileDropdown.classList.add('opacity-0', 'translate-y-2');
                    setTimeout(() => {
                        profileDropdown.classList.add('hidden');
                    }, 200); // Tiempo para que termine la transición
                }
            });

            // Cierra el dropdown al hacer clic fuera
            document.addEventListener('click', (e) => {
                if (!btnToggleProfile.contains(e.target) && !profileDropdown.contains(e.target) && !profileDropdown.classList.contains('hidden')) {
                     profileDropdown.classList.add('opacity-0', 'translate-y-2');
                     setTimeout(() => {
                        profileDropdown.classList.add('hidden');
                    }, 200);
                }
            });

            // Manejo de Desconexión (simulación de assets/js/Ajustes/Usuario.js)
            document.getElementById('btn-desconectar-sidebar').addEventListener('click', (e) => {
                e.preventDefault();
                if (window.Usuario && window.Usuario.handleLogout) {
                    window.Usuario.handleLogout();
                } else {
                    alertMessage("Función de desconexión no inicializada.", "error");
                }
            });
            document.getElementById('btn-desconectar-dropdown').addEventListener('click', (e) => {
                e.preventDefault();
                 if (window.Usuario && window.Usuario.handleLogout) {
                    window.Usuario.handleLogout();
                    // Oculta el dropdown después de la acción
                    profileDropdown.classList.add('opacity-0', 'translate-y-2', 'hidden');
                } else {
                    alertMessage("Función de desconexión no inicializada.", "error");
                }
            });
            
            // Inicia en la vista correcta basada en el hash o en Resumen Académico por defecto
            handleHashChange();

            // Lógica de navegación con teclado para el modal de búsqueda (opcional pero mejora UX)
            document.addEventListener('keydown', (e) => {
                // Ctrl+K o Cmd+K para abrir búsqueda
                if (e.key === 'k' && (e.ctrlKey || e.metaKey)) {
                    e.preventDefault();
                    openSearchModal();
                }
                // Escape para cerrar cualquier modal
                if (e.key === 'Escape') {
                    const activeModals = document.querySelectorAll('.modal:not(.hidden)');
                    activeModals.forEach(modal => {
                        modal.classList.add('hidden');
                    });
                }
            });
            
            // Renderiza los íconos de Lucide en toda la página
            if (typeof lucide !== 'undefined') {
                lucide.createIcons();
            }
        });
